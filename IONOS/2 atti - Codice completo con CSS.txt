## 🔷 **4. Integrazione nel codice dell’Atto 2**

```javascript
function updateAct2(t) {

    // Mostra scudo e WiFi
    shield.el.style.display = "block";
    wifiSymbol.style.display = "block";

    // Movimento orbitale (non modificato)
    updateShieldPositionFromEngine();
    updateYouPositionFromEngine();

    // Raggio corto SCAM → You$
    updateShortLaserScamToYou(t);

    // Deflessione sul WiFi
    updateWifiDeflection(t);

    // Flussi banche → You$ (non verso SCAM)
    const youPos = getYouPosition();
    tryCoinFlowBanksToYouOnly(t, youPos);

    // Ambulanza ferma fino a t=60
    if (t < 60) {
        ambulance.classList.remove("moving-to-scam", "moving-to-prison");
        return;
    }

    // Movimento ambulanza verso SCAM
    if (t >= 60 && t < 62.5) {
        ambulance.classList.add("moving-to-scam");
        moveAmbulanceToScam(t);

        // fade-out del truffatore
        fadeOutTruffatore();
        return;
    }

    // Movimento SCAM → Pianeta Prigione
    if (t >= 62.5 && t <= 65) {
        ambulance.classList.remove("moving-to-scam");
        ambulance.classList.add("moving-to-prison");
        moveAmbulanceToPrison(t);
        return;
    }
}
```
