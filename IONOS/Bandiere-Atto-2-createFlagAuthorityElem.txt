function getShortLaserVector() {
    const scamPos = getScamPosition();
    const youPos = getYouPosition();

    const dx = youPos.x - scamPos.x;
    const dy = youPos.y - scamPos.y;

    const dist = Math.sqrt(dx*dx + dy*dy);
    const half = dist * 0.5;

    const ux = dx / dist;
    const uy = dy / dist;

    return {
        x1: scamPos.x,
        y1: scamPos.y,
        x2: scamPos.x + ux * half,
        y2: scamPos.y + uy * half
    };
}
